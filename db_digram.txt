Table ecommerce.users as U {
  id int [pk, increment] 
  email varchar
  user varchar
  password varchar
  fname varchar
  lname varchar
  profilePic varchar
  detail text
  country_code int
  role_id int
  created_at datetime [default: `now()`]
  update_at timestamp
}

Table ecommerce.countries {
  code int [pk]
  name varchar
  continent_name varchar
  created_at datetime [default: `now()`]
 }

 Table ecommerce.role {
  id int [pk,increment]
  name varchar
  descirption varchar
  created_at datetime [default: `now()`]
 }

 Table ecommerce.seesion {
  id int [pk,increment]
  name varchar
  user_id int
  ip varchar
  created_at datetime [default: `now()`]
 }


Table ecommerce.order_items {
  order_id int [ref: > ecommerce.orders.id] 
  product_id int
  quantity int [default: 1] 
  created_at datetime [default: `now()`]
}

Ref: ecommerce.order_items.product_id > ecommerce.products.id

Table ecommerce.orders {
  id int [pk] 
  user_id int [not null, unique]
  status varchar
  created_at varchar [note: 'When order created'] 
}


Table ecommerce.products {
  id int [pk]
  name varchar
  merchant_id int [not null]
  price double
  created_at datetime [default: `now()`]
  

}
Table ecommerce.stock {
  id int [pk,increment]
  name varchar
  product_id int [not null]
  price int
  manage varchar
  quantity int [default: 1] 
  created_at datetime [default: `now()`]
}

Table ecommerce.product_tags {
  id int [pk]
  name varchar
}



Ref: ecommerce.product_tags.id <> ecommerce.products.id // many-to-many
//composite foreign key
